  • Simple Bar Chart

    import altair as alt
import pandas as pd
import streamlit as st
import numpy as np
from random import random

letters = [chr(l + 65) for l in range(26)]
values = [random() for i in letters]

df = pd.DataFrame({
    'a': letters,
    # 'b': np.random.randn(len(letters),1)
    'b': values
})

chart = alt.Chart(df).mark_bar().encode(
    x='a',
    y='b'
)

st.write(chart)

  • 2D Histogram Scatter Plot

    import pandas as pd
import numpy as np
import altair as alt
import streamlit as st


df = pd.DataFrame(
    np.random.randn(200, 3),columns=['a', 'b', 'c']
)

chart = alt.Chart(df).mark_circle().encode(
    x='a', y='b', size='c', color='c', tooltip=['a', 'b', 'c'])

st.write(chart)


    from vega_datasets import data
import streamlit as st
import altair as alt
import pandas as pd


chart = alt.Chart(data.movies.url).mark_circle().encode(
    alt.X('IMDB_Rating:Q').bin(),
    alt.Y('Rotten_Tomatoes_Rating:Q').bin(),
    size='count()'
)

st.write(chart)
